# Enumerate source files
FILE(GLOB SRC_FILES *.cpp *.inl)
FILE(GLOB INC_FILES *.h)

ADD_LIBRARY(BVHDependencies INTERFACE IMPORTED)

# CGAL
IF (TARGET lib_CGAL AND PYMESH_USE_CGAL)
    TARGET_LINK_LIBRARIES(BVHDependencies
        INTERFACE
            lib_CGAL
    )
    ADD_SUBDIRECTORY(CGAL)
ENDIF (TARGET lib_CGAL AND PYMESH_USE_CGAL)

# GeoGram
IF (TARGET lib_Geogram AND PYMESH_USE_GEOGRAM)
    TARGET_LINK_LIBRARIES(BVHDependencies
        INTERFACE
            lib_Geogram
    )
    ADD_SUBDIRECTORY(Geogram)
ENDIF (TARGET lib_Geogram AND PYMESH_USE_GEOGRAM)

# IGL
IF (TARGET lib_IGL AND PYMESH_USE_LIBIGL)
    TARGET_LINK_LIBRARIES(BVHDependencies
        INTERFACE
            lib_IGL
    )
    ADD_SUBDIRECTORY(IGL)
ENDIF (TARGET lib_IGL AND PYMESH_USE_LIBIGL)

ADD_LIBRARY(lib_BVH SHARED ${SRC_FILES} ${INC_FILES})
SET_TARGET_PROPERTIES(lib_BVH PROPERTIES OUTPUT_NAME "PyMesh-BVH")
TARGET_LINK_LIBRARIES(lib_BVH
    PUBLIC
        Mesh
        PyMesh::Tools
    PRIVATE
        BVHDependencies
)
